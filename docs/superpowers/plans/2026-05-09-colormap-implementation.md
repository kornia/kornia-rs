**Goal:** Add `apply_colormap` to `kornia-rs` covering all 21 OpenCV colormaps, with NEON/AVX2 SIMD kernels, PyO3 Python bindings, and a benchmark showing ≥10x speedup over `cv2.applyColorMap` on Jetson Orin.

**Architecture:** Pure-Rust LUT tables (3 × 256 bytes per colormap, channels separate) generated from OpenCV via a Python script to guarantee pixel-identical output. Runtime dispatch via the existing `CpuFeatures` singleton: NEON (aarch64, 16 px/iter via `vqtbl4q_u8`) → AVX2 (x86_64, 8 px/iter via `_mm256_i32gather_epi32`) → scalar fallback.

**Tech Stack:** Rust (`std::arch::aarch64`, `std::arch::x86_64`), PyO3, Criterion, NumPy, OpenCV-Python (`cv2`)

```rust
/// Apply a colormap to a grayscale u8 image, producing an RGB u8 image.
///
/// Input: `Image<u8, 1>` (single-channel grayscale, values 0–255).
/// Output: `Image<u8, 3>` (RGB, same spatial dimensions).
pub fn apply_colormap(
    src: &Image<u8, 1, CpuAllocator>,
    dst: &mut Image<u8, 3, CpuAllocator>,
    colormap: ColormapType,
) -> Result<(), ImageError> {
    if src.size() != dst.size() {
        return Err(ImageError::InvalidImageSize(
            src.cols(), src.rows(), dst.cols(), dst.rows(),
        ));
    }
    let lut = colormap.lut();
    apply_colormap_dispatch(lut, src.as_slice(), dst.as_slice_mut());
    Ok(())
}

#[inline]
fn apply_colormap_dispatch(lut: &ColormapLut, src: &[u8], dst: &mut [u8]) {
    #[cfg(target_arch = "aarch64")]
    {
        apply_colormap_neon(lut, src, dst);
        return;
    }
    #[cfg(target_arch = "x86_64")]
    {
        let cpu = crate::simd::cpu_features();
        if cpu.has_avx2 {
            unsafe { apply_colormap_avx2(lut, src, dst) };
            return;
        }
    }
    #[allow(unreachable_code)]
    apply_colormap_scalar(lut, src, dst);
}

pub(crate) fn apply_colormap_scalar(lut: &ColormapLut, src: &[u8], dst: &mut [u8]) {
    for (pixel, &idx) in dst.chunks_exact_mut(3).zip(src.iter()) {
        let i = idx as usize;
        pixel[0] = lut.r[i];
        pixel[1] = lut.g[i];
        pixel[2] = lut.b[i];
    }
}
```

```rust
#[cfg(target_arch = "aarch64")]
fn apply_colormap_neon(lut: &ColormapLut, src: &[u8], dst: &mut [u8]) {
    use std::arch::aarch64::*;
    let n = src.len();
    let sp = src.as_ptr();
    let dp = dst.as_mut_ptr();

    // Load each 256-byte channel table as 4 × vld1q_u8_x4 = 4 × 64-byte chunks.
    // vqtbl4q_u8 returns 0 for any index >= 64, so OR-ing 4 results
    // with offsets 0/64/128/192 covers the full [0, 255] range.
    let (rt0, rt1, rt2, rt3);
    let (gt0, gt1, gt2, gt3);
    let (bt0, bt1, bt2, bt3);
    unsafe {
        rt0 = vld1q_u8_x4(lut.r.as_ptr().add(0));
        rt1 = vld1q_u8_x4(lut.r.as_ptr().add(64));
        rt2 = vld1q_u8_x4(lut.r.as_ptr().add(128));
        rt3 = vld1q_u8_x4(lut.r.as_ptr().add(192));

        gt0 = vld1q_u8_x4(lut.g.as_ptr().add(0));
        gt1 = vld1q_u8_x4(lut.g.as_ptr().add(64));
        gt2 = vld1q_u8_x4(lut.g.as_ptr().add(128));
        gt3 = vld1q_u8_x4(lut.g.as_ptr().add(192));

        bt0 = vld1q_u8_x4(lut.b.as_ptr().add(0));
        bt1 = vld1q_u8_x4(lut.b.as_ptr().add(64));
        bt2 = vld1q_u8_x4(lut.b.as_ptr().add(128));
        bt3 = vld1q_u8_x4(lut.b.as_ptr().add(192));
    }

    let off64  = unsafe { vdupq_n_u8(64) };
    let off128 = unsafe { vdupq_n_u8(128) };
    let off192 = unsafe { vdupq_n_u8(192) };

    let bulk = n & !15; // round down to multiple of 16
    let mut i = 0usize;

    unsafe {
        while i < bulk {
            let idx = vld1q_u8(sp.add(i));

            // R channel: 4-chunk OR covers idx ∈ [0, 255]
            // vsubq_u8 wraps (unsigned), so out-of-chunk indices become ≥64 → vqtbl4q_u8 returns 0.
            let r = vorrq_u8(
                vorrq_u8(vqtbl4q_u8(rt0, idx),
                         vqtbl4q_u8(rt1, vsubq_u8(idx, off64))),
                vorrq_u8(vqtbl4q_u8(rt2, vsubq_u8(idx, off128)),
                         vqtbl4q_u8(rt3, vsubq_u8(idx, off192))),
            );
            let g = vorrq_u8(
                vorrq_u8(vqtbl4q_u8(gt0, idx),
                         vqtbl4q_u8(gt1, vsubq_u8(idx, off64))),
                vorrq_u8(vqtbl4q_u8(gt2, vsubq_u8(idx, off128)),
                         vqtbl4q_u8(gt3, vsubq_u8(idx, off192))),
            );
            let b = vorrq_u8(
                vorrq_u8(vqtbl4q_u8(bt0, idx),
                         vqtbl4q_u8(bt1, vsubq_u8(idx, off64))),
                vorrq_u8(vqtbl4q_u8(bt2, vsubq_u8(idx, off128)),
                         vqtbl4q_u8(bt3, vsubq_u8(idx, off192))),
            );
            // Write 16 interleaved RGB pixels in one instruction (48 bytes).
            vst3q_u8(dp.add(i * 3), uint8x16x3_t(r, g, b));
            i += 16;
        }
        // Scalar tail for remaining 0–15 pixels.
        while i < n {
            let idx = *sp.add(i) as usize;
            *dp.add(i * 3)     = lut.r[idx];
            *dp.add(i * 3 + 1) = lut.g[idx];
            *dp.add(i * 3 + 2) = lut.b[idx];
            i += 1;
        }
    }
}
```

```rust
#[cfg(target_arch = "x86_64")]
#[target_feature(enable = "avx2")]
unsafe fn apply_colormap_avx2(lut: &ColormapLut, src: &[u8], dst: &mut [u8]) {
    use std::arch::x86_64::*;

    // Pack LUT into u32: 0x00RRGGBB per entry for gather-based lookup.
    // Built once per call; fits in L1 (1 KB).
    let mut lut_packed = [0u32; 256];
    for i in 0..256 {
        lut_packed[i] = (lut.r[i] as u32) << 16
                      | (lut.g[i] as u32) << 8
                      | (lut.b[i] as u32);
    }
    let lut_ptr = lut_packed.as_ptr() as *const i32;

    let sp = src.as_ptr();
    let dp = dst.as_mut_ptr();
    let n  = src.len();
    let bulk = n & !7; // 8 pixels per iteration
    let mut i = 0usize;

    while i < bulk {
        // Load 8 gray indices as u8, zero-extend to i32 for gather.
        let idx_u8  = _mm_loadl_epi64(sp.add(i) as *const __m128i);
        let idx_i32 = _mm256_cvtepu8_epi32(idx_u8);

        // Gather 8 packed RGB values (scale=4: each index selects a u32).
        let packed = _mm256_i32gather_epi32(lut_ptr, idx_i32, 4);

        // Store gathered u32 values to a stack buffer, then write interleaved RGB.
        let mut buf = [0u32; 8];
        _mm256_storeu_si256(buf.as_mut_ptr() as *mut __m256i, packed);

        for j in 0..8 {
            let rgb = buf[j];
            *dp.add((i + j) * 3)     = ((rgb >> 16) & 0xFF) as u8; // R
            *dp.add((i + j) * 3 + 1) = ((rgb >> 8)  & 0xFF) as u8; // G
            *dp.add((i + j) * 3 + 2) = (rgb          & 0xFF) as u8; // B
        }
        i += 8;
    }
    // Scalar tail.
    while i < n {
        let idx = *sp.add(i) as usize;
        *dp.add(i * 3)     = lut.r[idx];
        *dp.add(i * 3 + 1) = lut.g[idx];
        *dp.add(i * 3 + 2) = lut.b[idx];
        i += 1;
    }
}
```

```rust
    Ok(out)
}
```

```rust
// crates/kornia-imgproc/benches/bench_colormap.rs
use criterion::{criterion_group, criterion_main, BenchmarkId, Criterion};
use kornia_image::{Image, ImageSize};
use kornia_imgproc::color::{apply_colormap, ColormapType};
use kornia_tensor::CpuAllocator;

fn make_images(w: usize, h: usize) -> (Image<u8, 1, CpuAllocator>, Image<u8, 3, CpuAllocator>) {
    let src = Image::<u8, 1, _>::from_size_val(
        ImageSize { width: w, height: h }, 128u8, CpuAllocator,
    ).unwrap();
    let dst = Image::<u8, 3, _>::from_size_val(
        ImageSize { width: w, height: h }, 0u8, CpuAllocator,
    ).unwrap();
    (src, dst)
}

fn bench_colormap_sizes(c: &mut Criterion) {
    let mut group = c.benchmark_group("colormap");
    group.sample_size(50);

    for (w, h) in [(640usize, 480usize), (1920, 1080), (3840, 2160)] {
        let label = format!("{w}x{h}");
        let (src, mut dst) = make_images(w, h);
        let lut = ColormapType::Jet;

        // Dispatch path (what users call — hits NEON on Jetson, AVX2 on x86)
        group.bench_with_input(
            BenchmarkId::new("dispatch/jet", &label),
            &label,
            |b, _| b.iter(|| apply_colormap(&src, &mut dst, lut).unwrap()),
        );

        // Scalar path — explicit, bypasses SIMD dispatch
        group.bench_with_input(
            BenchmarkId::new("scalar/jet", &label),
            &label,
            |b, _| {
                use kornia_imgproc::color::colormap::apply_colormap_scalar;
                let l = lut.lut();
                b.iter(|| apply_colormap_scalar(l, src.as_slice(), dst.as_slice_mut()));
            },
        );
    }
    group.finish();
}

criterion_group!(benches, bench_colormap_sizes);
criterion_main!(benches);
```

Note: `apply_colormap_scalar` and `ColormapType::lut()` must be `pub(crate)` — verify in `colormap.rs`. Change `pub(crate)` to `pub` on both if needed for the bench crate, or use `#[cfg(bench)]` visibility.

```python
# examples/bench_colormap.py
"""
Benchmark kornia-rs apply_colormap vs cv2.applyColorMap.

Run: python examples/bench_colormap.py
Requires: pip install opencv-python numpy kornia-rs (maturin develop already run)
"""
import timeit
import numpy as np
import cv2
import kornia_rs as kornia

COLORMAPS = [
    ("autumn",    cv2.COLORMAP_AUTUMN),
    ("bone",      cv2.COLORMAP_BONE),
    ("jet",       cv2.COLORMAP_JET),
    ("winter",    cv2.COLORMAP_WINTER),
    ("rainbow",   cv2.COLORMAP_RAINBOW),
    ("ocean",     cv2.COLORMAP_OCEAN),
    ("summer",    cv2.COLORMAP_SUMMER),
    ("spring",    cv2.COLORMAP_SPRING),
    ("cool",      cv2.COLORMAP_COOL),
    ("hsv",       cv2.COLORMAP_HSV),
    ("pink",      cv2.COLORMAP_PINK),
    ("hot",       cv2.COLORMAP_HOT),
    ("parula",    cv2.COLORMAP_PARULA),
    ("magma",     cv2.COLORMAP_MAGMA),
    ("inferno",   cv2.COLORMAP_INFERNO),
    ("plasma",    cv2.COLORMAP_PLASMA),
    ("viridis",   cv2.COLORMAP_VIRIDIS),
    ("cividis",   cv2.COLORMAP_CIVIDIS),
    ("twilight",  cv2.COLORMAP_TWILIGHT),
    ("turbo",     cv2.COLORMAP_TURBO),
    ("deepgreen", cv2.COLORMAP_DEEPGREEN),
]

SIZES   = [(640, 480), (1920, 1080), (3840, 2160)]
REPEATS = 200

print(f"{'Colormap':12s}  {'Size':12s}  {'cv2 ms':>8s}  {'kornia ms':>9s}  {'Speedup':>8s}  {'MP/s':>8s}")
print("-" * 72)

for w, h in SIZES:
    gray = np.random.randint(0, 256, (h, w), dtype=np.uint8)
    total_px = w * h

    for name, cv2_id in COLORMAPS:
        # Warmup
        cv2.applyColorMap(gray, cv2_id)
        kornia.imgproc.apply_colormap(gray, name)

        t_cv2    = timeit.timeit(lambda: cv2.applyColorMap(gray, cv2_id), number=REPEATS)
        t_kornia = timeit.timeit(lambda: kornia.imgproc.apply_colormap(gray, name), number=REPEATS)

        speedup = t_cv2 / t_kornia
        mpps    = (total_px * REPEATS) / t_kornia / 1e6

        print(f"{name:12s}  {w}x{h:4d}    {t_cv2*1000/REPEATS:8.2f}  {t_kornia*1000/REPEATS:9.2f}  {speedup:7.1f}x  {mpps:7.1f}")
    print()
```

## Self-Review Notes

1. `ColormapType::lut()` is called in both the Rust kernel and the bench. Visibility changed to `pub` in Task 7 — verify no breakage with `cargo check -p kornia-imgproc`.
2. The `apply_colormap_avx2` function signature uses `unsafe fn` — the `apply_colormap_dispatch` caller wraps it in `unsafe {}`. Confirm this compiles on aarch64 (AVX2 branch is `#[cfg(target_arch = "x86_64")]` so it's excluded from the aarch64 build).
3. The Python smoke test in Task 6 compares BGR→RGB flipped cv2 output. The generator script also extracts R from channel 2. These must be consistent — both swap BGR→RGB. ✓
4. `maturin develop` without `--release` produces a debug build with ~5× lower SIMD throughput. Always use `--release` for benchmark runs.
